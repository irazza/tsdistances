X = randn(100, 200);           % 100 time series, each length 200

% Compute all pairwise DTW distances
D = tsd_dtw(X);                % Returns 100x100 distance matrix

% Find closest pair to first time series
[dist, idx] = min(D(1, 2:end));
disp(['Closest match: series ', num2str(idx+1)]);
disp(['Distance: ', num2str(dist)]);
